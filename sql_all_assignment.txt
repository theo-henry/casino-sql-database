  PRIMARY KEY (Gender_code)
);

CREATE TABLE KYC_Document_Type (

  Status_code  CHAR(1)   NOT NULL,
  Description  VARCHAR(40) NOT NULL,
  PRIMARY KEY (Status_code)
);

CREATE TABLE Address_Type (
  Type_ID    INT     NOT NULL,
  Description  VARCHAR(40) NOT NULL,
  PRIMARY KEY (Type_ID)
);

CREATE TABLE Currency (
  Currency_ID  CHAR(3)   NOT NULL,
  Currency_Code SMALLINT(6) NOT NULL,
  Description  VARCHAR(40) NOT NULL,
  PRIMARY KEY (Currency_ID)
);

CREATE TABLE CC_Payment_Type (
  CCType    CHAR(2)   NOT NULL,
  Description  VARCHAR(40) NOT NULL,
  PRIMARY KEY (CCType)
);

CREATE TABLE CC_Payment_State (
  CCState    CHAR(1)   NOT NULL,
  Description  VARCHAR(40) NOT NULL,
  PRIMARY KEY (CCState)
);

CREATE TABLE ZIP_code (
  ZIP_code INT     NOT NULL,
  Country  VARCHAR(20) NOT NULL,
  City   VARCHAR(20) NOT NULL,
  State   VARCHAR(20) NOT NULL,
  PRIMARY KEY (ZIP_code, Country)
);

CREATE TABLE User_Address (
  Address_ID   BIGINT(20)  NOT NULL,
  Type_ID     INT     NOT NULL,
  Address_Number SMALLINT(6) NOT NULL,
  ZIP_code    INT     NOT NULL,
  Country     VARCHAR(20) NOT NULL,
  PRIMARY KEY (Address_ID),
  CONSTRAINT fk_user_address_type
    FOREIGN KEY (Type_ID) REFERENCES Address_Type(Type_ID),
  CONSTRAINT fk_user_address_zip
    FOREIGN KEY (ZIP_code, Country) REFERENCES ZIP_code(ZIP_code, Country)
);

CREATE TABLE Registered_User (
  User_ID     INT      NOT NULL,
  Status_code   CHAR(1)    NOT NULL,
  Username    VARCHAR(64)  NOT NULL,
  First_name   VARCHAR(64)  NOT NULL,
  Last_name    VARCHAR(64)  NOT NULL,
  Second_surname VARCHAR(64),
  Email      VARCHAR(64)  NOT NULL,
  Phone_number  BIGINT(20),
  Address_ID   BIGINT(20),
  Passkey    VARCHAR(64)  NOT NULL,
  Date_of_birth  DATE     NOT NULL,
  Gender_code   CHAR(1)    NOT NULL,
  Registration_date TIMESTAMP  NOT NULL,
  Currency_ID   CHAR(3)    NOT NULL,
  Account_balance DECIMAL(12,2),
  PRIMARY KEY (User_ID),
  UNIQUE KEY uq_registered_user_username (Username),
  UNIQUE (Email),
  CONSTRAINT fk_registered_user_status
    FOREIGN KEY (Status_code)  REFERENCES User_Status(Status_code),
  CONSTRAINT fk_registered_user_gender
    FOREIGN KEY (Gender_code)  REFERENCES User_Gender(Gender_code),
  CONSTRAINT fk_registered_user_address
    FOREIGN KEY (Address_ID)  REFERENCES User_Address(Address_ID),
  CONSTRAINT fk_registered_user_currency
    FOREIGN KEY (Currency_ID)  REFERENCES Currency(Currency_ID)
);

CREATE TABLE KYC_Document (
  KYC_doc_ID     BIGINT(20) NOT NULL,
  User_ID       INT     NOT NULL,
  Doc_type_code    CHAR(3)   NOT NULL,
  Document_number   VARCHAR(100) NOT NULL,
  Issuing_country   CHAR(2),
  Issue_date     DATE,
  Expiry_date     DATE,
  Status_code     CHAR(1)   NOT NULL DEFAULT 'P',
  Submitted_timestamp TIMESTAMP  NOT NULL,
  Verified_timestamp TIMESTAMP,
  Last_updated    TIMESTAMP,
  PRIMARY KEY (KYC_doc_ID),
  CONSTRAINT fk_kyc_document_user
    FOREIGN KEY (User_ID)    REFERENCES Registered_User(User_ID),
  CONSTRAINT fk_kyc_document_type
    FOREIGN KEY (Doc_type_code) REFERENCES KYC_Document_Type(Doc_type_code),
  CONSTRAINT fk_kyc_document_status
    FOREIGN KEY (Status_code)  REFERENCES KYC_Verification_Status(Status_code)
);

CREATE TABLE Game_Type (
  Game_type_code CHAR(3)   NOT NULL,

  PRIMARY KEY (Game_type_code)
);

CREATE TABLE Game (
  Game_ID       INT      NOT NULL,
  Game_name      VARCHAR(100)  NOT NULL,
  Game_type_code    CHAR(3)    NOT NULL,
  Min_bet_amount    DECIMAL(12,2) NOT NULL,
  Max_bet_amount    DECIMAL(12,2) NOT NULL,
  PRIMARY KEY (Game_ID),
  CONSTRAINT fk_game_type
    FOREIGN KEY (Game_type_code) REFERENCES Game_Type(Game_type_code)
);

CREATE TABLE Game_Round (
  Round_ID          BIGINT(20) NOT NULL,
  Game_ID          INT     NOT NULL,
  Start_time         TIMESTAMP  NOT NULL,
  End_time          TIMESTAMP  NOT NULL,
  Random_number_generator_seed INT    NOT NULL,
  PRIMARY KEY (Round_ID),
  CONSTRAINT fk_game_round_game
    FOREIGN KEY (Game_ID) REFERENCES Game(Game_ID)
);

CREATE TABLE Bet_outcome_type (
  Outcome_code CHAR(1)   NOT NULL,
  Description  VARCHAR(40) NOT NULL,
  PRIMARY KEY (Outcome_code)
);

CREATE TABLE Bet (
  Bet_ID    BIGINT(20)  NOT NULL,
  Round_ID   BIGINT(20)  NOT NULL,
  User_ID    INT     NOT NULL,
  Bet_amount  DECIMAL(12,2) NOT NULL,
  Bet_time   TIMESTAMP  NOT NULL,
  Outcome_code CHAR(1)   NOT NULL,
  Ticket_ID   BIGINT(20)  NOT NULL,
  PRIMARY KEY (Bet_ID),
  CONSTRAINT fk_bet_round
    FOREIGN KEY (Round_ID)   REFERENCES Game_Round(Round_ID),
  CONSTRAINT fk_bet_user
    FOREIGN KEY (User_ID)   REFERENCES Registered_User(User_ID),
  CONSTRAINT fk_bet_outcome
    FOREIGN KEY (Outcome_code) REFERENCES Bet_outcome_type(Outcome_code)
);

CREATE TABLE Bet_payout (
  Bet_ID      BIGINT(20)  NOT NULL,
  Payout_amount  DECIMAL(12,2) NOT NULL,
  Payout_timestamp TIMESTAMP  NOT NULL,
  PRIMARY KEY (Bet_ID),
  CONSTRAINT fk_bet_payout_bet
    FOREIGN KEY (Bet_ID) REFERENCES Bet(Bet_ID)
);

CREATE TABLE Ticket (

  Currency_ID   CHAR(3)   NOT NULL,
  Ticket_timestamp TIMESTAMP  NOT NULL,

    FOREIGN KEY (Currency_ID) REFERENCES Currency(Currency_ID),

3) Include an explanation of your model

Here is a description of the logical reasoning behind our data model. This explanation follows the chronological design process we used to construct a compliant, scalable, and normalized online casino database.

The Registered User

We began our design process by identifying the central entity of our system which is the player. We created the registered_user table to serve as the anchor for all other relationships. When defining the attributes for this table, we immediately applied normalization principles to ensure data integrity.
We recognized that storing status descriptions (like "Active" or "Banned") and gender descriptions directly in the user table would create redundant data and risk inconsistencies. To solve this and satisfy Third Normal Form, we extracted these attributes into separate lookup tables: user_status and user_gender. This decision ensures that any change to a status description happens in one place and propagates system-wide without needing to update millions of user records.

Handling Location and Address Logic

Next, we addressed how to store user locations. We realized that a simple text field for an address was insufficient for a real-world application that requires precise geolocation for legal compliance. We created the user_address table to separate location data from account data.
To strictly adhere to Third Normal Form, we analyzed the dependencies within an address. We determined that a City and State are functionally dependent on the ZIP Code, not strictly on the specific address ID. If we stored "New York" and "NY" in every single address row, we would have a transitive dependency. We resolved this by creating a dedicated zip_code table. Now, the user_address table only refers to a ZIP code, and the zip_code table acts as the single source of truth for the City, State, and Country associated with that code.

Security and Compliance (KYC)

Operating an online casino requires strict adherence to Know Your Customer (KYC) laws. We knew we needed to store identity documents, but adding columns to the user table would be inflexible. A user might upload a passport today and a driver’s license next year. To handle this, we designed the kyc_document table as a distinct entity linked to the user.
We further normalized this by creating kyc_document_type (to standardize inputs like Passport or ID Card) and kyc_verification_status (to track the approval workflow). This structure provides a clear audit trail. We can see exactly when a document was submitted, who verified it, and its current validity without cluttering the main user profile.

Structuring the Financial Ledger

We then moved to the most critical aspect of the business which is money management. We made a deliberate choice to decouple "banking" from "gaming." We introduced the ticket table to serve as the general ledger for the user's wallet. This table records distinct financial events like deposits and withdrawals.
For the actual processing of funds, we designed the cc_payment table. We noticed that credit card details (like the encrypted number and bank name) are attributes of the card used, not the transaction itself. To maintain strict data separation and security, we moved these details into cc_payment_card. This ensures that cc_payment remains a lightweight record of the transaction success or failure, while the sensitive card data is isolated in its own table. 

Constructing the Game Engine

Finally, we designed the core product offering. We established a hierarchy starting with game_type to categorize our inventory into Slots, Blackjack, Roulette, Poker, and more...The game table holds the specific configuration for each title, such as minimum and maximum bet limits.
We recognized that a "Bet" is not just an isolated action but occurs within a specific context. Therefore, we introduced the game_round entity. A round represents a single spin of a slot machine or a single hand of blackjack. This table captures the start and end times and, crucially, the Random Number Generator seed. This is vital for real-world fairness auditing.
The bet table links the user to a specific game_round. We purposely decoupled bet from the ticket table to optimize performance, as high-frequency betting would overwhelm a banking ledger. Lastly, we separated the outcome into a bet_payout table. This allows us to record a bet as a financial inflow immediately while the payout acts as a conditional financial outflow. This separation handles losses gracefully (no row in the payout table) and simplifies the calculation of Gross Gaming Revenue.

SELECT g.Game_name, COUNT(b.Bet_ID) AS Total_Bets_Placed
FROM Bet b
JOIN Game_Round gr ON b.Round_ID = gr.Round_ID
JOIN Game g ON gr.Game_ID = g.Game_ID
GROUP BY g.Game_name
ORDER BY Total_Bets_Placed DESC
LIMIT 3;

FROM Bet b
JOIN Game_Round gr ON b.Round_ID = gr.Round_ID
JOIN Game g ON gr.Game_ID = g.Game_ID
WHERE b.Outcome_code = 'W'
GROUP BY g.Game_name

SELECT DATE_FORMAT(Transaction_Date, '%Y-%m') AS Month, SUM(Cash_Flow) AS Net_Revenue
FROM (
  SELECT Bet_time AS Transaction_Date, Bet_amount AS Cash_Flow
  FROM Bet
  UNION ALL
  SELECT Payout_timestamp AS Transaction_Date, -Payout_amount AS Cash_Flow
  FROM Bet_payout
